====== audio ======

mac input.ape output.wav -d
flac --best input.wav output.flac
lame -b 128 input.wav output.mp3

cue2tracks -R -c flac -o “/path/%t-%P” name.cue

shntool conv *.ape -o flac
shntool split -f CDImage.cue -t '%n.%p-%t' -o flac CDImage.ape
