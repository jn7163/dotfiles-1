====== openssh ======

#upload
scp localpath/file username@hostname:/remotepath

#download
scp username@hostname:/remotepath/file localpath/file

#proxy
ssh -N -D 9090 username@hostname -p port

#mount
sshfs -p port virtio@localhost:/remotepath localpath

#umount
fusermount -u localpath
