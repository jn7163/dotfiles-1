====== smbclient ======

smbclient -L IP			    		#查看共享目录
smbclient //IP/share			    	#登录目标目录文档
smbclient //IP/share -U username -P password	#用户登录目录文档
smbclient //IP/share -U Anonymous -N	    	#匿名登录目录文档

#put和get进行资源存取，cd进目录，ls查看


mount.cifs //IP/share /local/dir		#挂载共享目录至本地
smbmount //IP/share /local/dir			#挂载共享目录至本地
