====== iptables ======

eth0:10.170.11.168  internet
eth1:192.168.10.1   workgroup
eth2		    internet

echo 1 > /proc/sys/net/ipv4/ip_forward
iptables -A POSTROUTING -t nat -s 102.168.10.0/24 -j SNAT --to-source 10.170.11.168
iptables -t nat -A POSTROUTING -o eth0 -j MASQUERADE
iptables -t nat -A POSTROUTING -o eth2 -j MASQUERADE


echo "1" > /proc/sys/net/ipv4/ip_forward
ifconfig eth0 0.0.0.0 up
ifconfig eth1 0.0.0.0 up
brctl addbr br0
brctl addif br0 eth0
brctl addif br0 eth1
ifconfig br0 192.168.1.1 netmask 255.255.255.0 broadcast 192.168.1.255 up
route add default gw 192.168.1.254
