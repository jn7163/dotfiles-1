====== Ubuntu ======


$ apt-get update					#更新源
$ apt-get upgrade					#安装可用更新
$ apt-cache search package			#查询软件包
$ apt-cache rdepends pachage		#查看软件包的依赖
$ apt-cache show package			#显示软件包的完整描述
$ apt-get install package				#安装一个软件包
$ apt-get remove package			#卸载一个软件包
$ apt-get help						#列举其他apt-get命令
$ apt-get dist-upgrade				#升级系统
$ apt-get dselect-upgrade			#使用dselect升级
$ apt-get build-dep package			#安装软件包的编译环境
$ apt-get source package				#下载软件包的源代码
$ apt-get clean && apt-get autoclean	#清理下载文件的存档和清理过时的包
$ apt-get check package				#检查是否有损坏的依赖

$ dpkg -i package					#本地安装deb软件包
$ dpkg -c package					#列出本地安装包的内容
$ dpkg -I package					#从本地deb包中提取包裹的信息
$ dpkg -r package					#移除一个已安装的包裹
$ dpkg -P package					#完全清除一个已安装的包裹，包括配置文件
$ dpkg -L package					#列出本地deb安装包安装的所有文件清单
$ dpkg -s package					#显示已安装包裹的信息
$ dpkg-reconfigure package			#重新配置一个已安装的包裹
